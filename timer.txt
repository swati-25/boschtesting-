#define TIMER0_H 0x19
#define TIMER0_L 0x19
#define LED      P10
#define LED_01   P11

short i=0;

void timer0_interrupt() interrupt 1
{
	if(i== 4000)
	{
		LED = !LED ;
		i=0;
	}
	else
		i++;
}

{

}
